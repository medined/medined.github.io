--- layout: post title: Spring Boot Hello World Web Turtial (or Getting Started with spring-boot-starter-web) date: '2014-12-25T11:25:00.000-05:00' author: David Medinets tags: - Spring Boot modified_time: '2014-12-25T11:25:12.166-05:00' blogger_id: tag:blogger.com,1999:blog-3207985.post-8690011422079695119 blogger_orig_url: https://www.blogger.com/comment.g?blogID=3207985&postID=8690011422079695119 --- There are several tutorials available showing how to start using Spring Boot. However, they seemed incomplete and they mostly used Gradle. Since I prefer Maven, I am making my own tutorial.

The first step is to create your directory structure.

cd {project directory}
mkdir -p src/main/java/com/codebits/ragnvald

Now create the pom.xml in the project's root directory.

<project xmlns="http://maven.apache.org/POM/4.0.0" xmlns:xsi="http://www.w3.org/2001/XMLSchema-instance" xsi:schemaLocation="http://maven.apache.org/POM/4.0.0 http://maven.apache.org/xsd/maven-4.0.0.xsd">
    <modelVersion>4.0.0</modelVersion>

    <groupId>com.codebits</groupId>
    <artifactId>ragnvald</artifactId>
    <version>0.0.1-SNAPSHOT</version>
    <packaging>jar</packaging>

    <name>ragnvald</name>
    <url>http://www.codebits.com</url>

    <parent>
        <groupId>org.springframework.boot</groupId>
        <artifactId>spring-boot-starter-parent</artifactId>
        <version>1.2.0.RELEASE</version>
    </parent>

    <properties>
        <project.build.sourceEncoding>UTF-8</project.build.sourceEncoding>
        <start-class>com.codebits.ragnvald.Application</start-class>
        <maven.compiler.source>1.8</maven.compiler.source>
        <maven.compiler.target>1.8</maven.compiler.target>
    </properties>

    <dependencies>
        <dependency>
            <groupId>org.springframework.boot</groupId>
            <artifactId>spring-boot-starter-web</artifactId>
        </dependency>        
    </dependencies>
  
    <build>
        <plugins>
            <plugin>
                <groupId>org.apache.maven.plugins</groupId>
                <artifactId>maven-shade-plugin</artifactId>
                <version>2.3</version>
                <executions>
                    <execution>
                        <phase>package</phase>
                        <goals>
                            <goal>shade</goal>
                        </goals>
                    </execution>
                </executions>
            </plugin>
        </plugins>
    </build>

</project>

Next create the src/main/java/com/codebits.ragnvald.HelloController.java file.

package com.codebits.ragnvald;

import org.springframework.web.bind.annotation.RestController;
import org.springframework.web.bind.annotation.RequestMapping;

@RestController
public class HelloController {

    @RequestMapping("/")
    public String index() {
        return "HelloWorld!";
    }

}

Now create the src/main/java/com/codebits.ragnvald.Application.java file.

package com.codebits.ragnvald;

import java.util.Arrays;

import org.springframework.boot.SpringApplication;
import org.springframework.boot.autoconfigure.EnableAutoConfiguration;
import org.springframework.context.ApplicationContext;
import org.springframework.context.annotation.ComponentScan;
import org.springframework.context.annotation.Configuration;

@Configuration
@EnableAutoConfiguration
@ComponentScan
public class Application {

    public static void main(String[] args) {
        ApplicationContext ctx = SpringApplication.run(Application.class, args);

        System.out.println("Let's inspect the beans provided by Spring Boot:");

        String[] beanNames = ctx.getBeanDefinitionNames();
        Arrays.sort(beanNames);
        for (String beanName : beanNames) {
            System.out.println(beanName);
        }
    }

}

Now compile

mvn package

And run

java -jar target/ragnvald-0.0.1-SNAPSHOT.jar

Visit http://localhost:8080 to see the Hello World message.
